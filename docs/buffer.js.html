'use strict';

const FileSystem = require('fs');
const Readline = require('readline');

/**
 * Read a file without charset specification, then obtain the length of the
 * resulting Buffer.
 *
 * @summary Read a file as a Buffer.
 * @param {string} file Path to the file.
 */
function getBuffer(file) {
  FileSystem.readFile(file, 'utf-8', (error, buffer) => {
    if (error) {
      console.log(`An error occured; #{error}`);
    }
    const array = buffer.split('\n');
    for (const line of array) {
      console.log(line + 'aaaaaaa');
    }
  });
}

/**
 * Create a stream to read a file and use a readline interface as controller,
 * then return that readline interface for usage.
 * Obtain a read stream from a file, then create the console interface and
 * return it after configuration.
 * @summary Obtain a readline interface to output from file.
 * @param {string} file Path of the file to be read.
 * @return {Readline} Console interface reading a file.
 */
function readLines(file) {
  const fileStream = FileSystem.createReadStream(file);
  console.log(fileStream.read());
  return Readline.createInterface({
    input: fileStream,
    ctrlDelay: Infinity,
  });
}

/**
 * Create a file stream, then a terminal controller for it (readline) and
 * proceed to log every line of the file.
 * @summary Log lines of `file`
 * @param {string} file Path of the file to be read.
 */
async function processLines(file) {
  const terminal = readLines('input.txt');

  for await (const line of terminal) {
    console.log(line);
  }
}
processLines('input.txt');
